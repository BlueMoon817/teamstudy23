1. Module

  • <script type="module">
    import 가져올거 from '경로'
    import Element from '경로'
    </script>
    모듈 파일에서 요소 내보내기
    export default Element;
    export default 는 파일당 한번만 사용가능.
    여러개 내보내고 싶으면 export로 선언해준다.
    1 ) export 내보낼거1
    export 내보낼거2
    2 ) export {내보낼거1, 내보낼거2}
    3 ) export var a=10
    export var b=10
    export default 로 내보냈을 경우 가져올 것의 export된 이름을 변경해서 작명해도 된다.
    export 로만 내보냈을 경우 가져올 것의 export된 이름을 동일하게 써서 중괄호에 담아준다.
    import는 import {a} from "경로"
    export 와 export default를 동시에 쓸수 있다.
    export 와 export default를 동시에 사용하여 내보내기 한 것을 가져올 경우
    import 작명,{a,b} from '경로';
    export로 내보내기한 것을 가져올 때 이름을 바꾸고 싶을 경우
    import {기존이름 as 작명} from "경로";
    모든 걸 다 import 해오는 * 기호 (export로 내보낸것만 가능)
    import * as 별명 from "경로";
    코드를 작성할 때
    별명.원래이름 으로 명시한다.
    // 옛날문법
    var 가져오기 = require('경로')
