7. Template literals , tagged literals 연습문제와 해설

  • tagged literal 해체분석기를 이용해서 단어순서를 바꾸려면 어떻게해야할까요?

    (해체분석기`바지${pants} 양말${socks}`;
          이렇게 코드를 실행하면 콘솔창에 `양말20 바지100`이라는 문장이 떠야합니다.)

    var pants = 20;
    var socks = 100;
    function 해체분석기(문자들, 변수들1, 변수들2){
      console.log(문자들[1]+변수들1+문자들[0]+변수들2);
    }
    해체분석기`바지${pants} 양말${socks}`;

  • 바지가 0개면 '바지다팔렸어요'라는 문자를 대신 표시해주기

    var pants = 0;
    var socks = 100;
    function 해체분석기(문자들, 변수들1, 변수들2){
      if(변수들1===0){
        console.log(`바지다팔렸어요 양말`+변수들2);
      }
    }
    해체분석기`바지${pants} 양말${socks}`;
