  • Symbol의 활용

    Object 자료형의 비밀스런 key값

    객체에 비밀스러운 값을 저장하고 싶을 때

    var 심볼 = Symbol('설명');
    var weight = Symbol('내 시크릿 몸무게');
    var weight = Symbol('내 키');
    var person = {
      name: 'yura',
      //Symbol을 직접넣기
      [age] : 100
    };
    // person에 weight라는 심볼의 값을 저장하기
    person[weight] = 100;
    person[height] = 160;
    console.log(person); // {name:"yura", Symbol(내 시크릿 몸무게): 100}
    //콘솔에서 찍는건 가능하지만 반복문에서만 나오지 않음
    for(var key in person){
      console.log(person[key]); // yura -> symbol은 나오지 않음.
      // for문에서 person[weight]가 출력되지 않는 이유는 enumerable하지 않기 때문
    }

  • Symbol의 특징

    설명이 같다고 같은 symbol이 아님

    var a = Symbol('설명1');
    var b = Symbol('설명1');
    // a와 b는 같지 않다.

    전역 심볼을 만들고 싶을 때

    Symbol.for()

    var a = Symbol.for('설명1');
    var b = Symbol.for('설명1');
    // Symbol.for()를 사용하여 설명을 작성하면 변수생성시 설명이 동일한 경우 기존 심볼을 복사해준다.
    // 즉 위의 b 변수의 선언은 var b=a와 같다.
    // 따라서 a == b 는 true이다.

  • 기본 내장 Symbol들

    var 어레이 = [2,3,4];
    //배열을 생성할 때 기본으로 들어가는 Symbol
    어레이[Symbol.iterator];
    // 어레이를 반복문에 돌릴시 어레이[Symbol.iterator]는 출력이 되지 않는다.
