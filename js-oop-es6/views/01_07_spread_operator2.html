2. 스프레드 연산자 (Spread Operator) 활용방법2 & apply, call 함수

  • 함수에 배열 인자 넣는 방법

    //배열의 모든 요소를 함수의 인자로 넣고 싶은 경우
    function 더하기(a,b,c){
      console.log(a+b+c);
    }
    var 어레이 = [10,20,30];

      1. 더하기(어레이[0],어레이[1],어레이[2]); // ..무지성방식,,,

      2. 더하기.apply(undefined, 어레이); // 옛날방식

           undefined에 적용해주세요 -> 그냥 더하기 함수를 실행하라는 의미.

           편법으로 편하게 인자로 넣기 위한 방법이었음.

      3. 더하기(...어레이); // 요즘 방법

  • person2에 정의되지 않은 person 객체의 인사 메소드를 사용하고 싶을 경우

    apply() , call() 사용

    var person={
      인사 : function(){
        console.log(this.name + '안녕');
      }
    }
    person.인사(); //undefined안녕

    var person2 = {
      name: "yura"
    }
    // person의 인사 메소드를 person2에 적용
    console.log(person.인사.apply(person2)); //yura안녕

  • 파라메터가 있는 경우

    person.인사.apply(person2, 인사함수에 주고 싶은 인자)

    console.log(person.인사.apply(person2, 1, 2))
    console.log(person.인사.apply(person2, [1, 2]))
    //apply는 배열로 인자를 넣을 수 있음(call과의 유일한 차이/ 기능은 같음)
    console.log(person.인사.call(person2, 1, 2));
