7. Class (ES6)

  • 상속기능을 구현하는 ES6 - class

    constructor 만드는 신문법

    자식.__proto__ == 부모.prototype

    Object.getPrototypeOf(자식);

    자식객체의 prototype 출력

    class 부모{
      constructor(파라미터){
        this.name = 파라미터;
        // 생성될 객체가 직접 가지는 함수
        this.sayHi = function(){
          console.log('Hi');
        }
      }
      // constructor 밖에 선언한 함수는 새로 생성될 객체가 물려받지 않는다.
      // 부모의 prototype에만 추가된다.
      sayHello(){
        console.log('Hello');
      }
    }

    var 자식 = new 부모();

    // 부모클래스의 프로토타입을 동적으로 추가하기
    부모.prototype.sayBye = function(){ console.log('bye~') }

  • class를 복사하는 extends / super

    유사한 class를 생성하고 싶을 때 사용.

    extends를 사용하여 존재하는 클래스를 확장하여 새 클래스를 만들었을 경우
              constructor의 첫줄에 super()를 써주어야 한다.

    class 할아버지{
      constructor(name1,name2){
        this.성 = 'Kim';
        this.이름1 = name1;
        this.이름2 = name2;
      }
      sayHi(){
        console.log('안녕');
      }
    }

    var 할아버지1 = new 할아버지('만덕');

    class 아버지 extends 할아버지{
      constructor(name1,name2,age){
        super(name1,name2);
        this.나이=age;
      }
      sayHi(){
        console.log('아버지');
        // 할아버지의 sayHi함수도 실행시키기     super.sayHi();   }
    }

    var 아버지1 = new 아버지('만수','만두',30);
    console.log(아버지1); // {성:"Kim", 이름1:"만수", 이름2:"만두", 나이:30}
    아버지1.sayHi(); // 아버지, 안녕

  • 연습문제

    class 할아버지{
      constructor(name){
        this.성 = 'Kim';
        this.이름 = name
      }
      sayHi(){
        console.log('안녕 나는 할아버지')
      }
    }

    class 아버지 extends 할아버지{
      constructor(name){
        super(name);
        this.나이 = 50;
      }
      sayHi2(){
        console.log('안녕 나는 아버지');
        super.sayHi();
      }
    }

    var a = new 아버지('만수');
