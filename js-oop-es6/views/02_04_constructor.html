4. Constructor(생성자) 의 용도

  • 비슷한 object를 쉽게 찍어낼 수 있는 constructor

    키가 동일한 여러개의 데이터객체를 만들고 싶을 때 사용할 수 있다.

    기본 유형을 만들어두고 값만 다르게 넣어서 복사하듯 생성한다.

    생성자 함수에서의 this는 새로 생성되는 object = instance이다.

    // 학생 객체처럼 name과 age를 가진 여러개의 객체를 생성하고 싶을 때,
    var 학생 = {name:"Kim", age:15}

    function 생성자함수(){
      this.name="Kim";
      this.age = 15;
    }

    var 학생1 = new 생성자함수();
    var 학생2 = new 생성자함수();
    console.log(학생1); // {name:"Kim", age:15}
    console.log(학생2); // {name:"Kim", age:15}

  • 생성자 함수로 생성한 객체에 메서드 추가하기

    // 생성자함수의 내장함수 추가
    function 생성자함수(){
      this.name="Kim";
      this.age = 15;
      this.sayHi=function(){
        console.log('안녕하세요' + this.name + '입니다');
      }
    }
    학생1.sayHi(); // 안녕하세요 Kim입니다.
    학생2.sayHi(); // 안녕하세요 Kim입니다.

  • 생성자함수로 생성한 객체의 속성값을 동적으로 넣기

    키의 값을 파라미터를 설정하여 인자로 받을 수 있다.

    // 생성자함수의 내장함수 추가
    function 생성자함수(name, age){
      this.name= name;
      this.age = age;
      this.sayHi=function(){
        console.log('안녕하세요' + this.name + '입니다');
      }
    }
    학생1.sayHi('Park'); // 안녕하세요 Park입니다.
    학생2.sayHi('Mun'); // 안녕하세요 Mun입니다.

  • 연습문제1

    쇼핑몰에 사용할 상품데이터를 constructor를 이용하여 만들기

    function dataProducts(product, price){
      this.product=product;
      this.price=price;
    }
    let product1 = new dataProducts('shirt',50000);
    let product2 = new dataProducts('pants',60000);
    console.log(product1, product2);

  • 연습문제2

    상품마다 부가세()라는 내부함수를 실행하면
              콘솔창에 상품가격 * 10%만큼의 부가세금액이 출력되도록 constructor 함수를 수정하기

    product1.부가세() 이렇게 쓰면 콘솔창에 상품의 부가세 값을 출력시키기

    function dataProducts(product, price){
      this.product=product;
      this.price=price;
      this.부가세=function(){
        console.log(this.price*0.1);
      }
    }

    let product1 = new dataProducts('shirt',50000);
    product1.부가세(); // 5000
