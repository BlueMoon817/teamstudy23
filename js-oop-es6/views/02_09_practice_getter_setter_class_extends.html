  • 1. 직접 class 구조만들어보기

    var 강아지1 = { type : '말티즈', color : 'white' };
    var 강아지2 = { type : '진돗개', color : 'brown' };

    위의 정보구성으로 class를 만들어서 오브젝트를 출력하기

    class 강아지{
      constructor(type, color){
        this.type=type;
        this.color=color;
      }
    }
    var 강아지1 = new 강아지('말티즈', 'white');
    var 강아지2 = new 강아지('진돗개', 'brown');
    console.log(강아지1);
    console.log(강아지2);

  • 2. 직접 class 구조만들어보기

    type,color는 강아지object와 비슷한데 age속성만 추가하고 싶은 경우.

    extends 사용해서 클래스 생성해보기

    class 고양이 extends 강아지{

    var 고양이2 = new 고양이('러시안블루', 'brown', 2);

    console.log(고양이2);

  • 3. 고양이와 강아지 object들에 기능 추가하기

    모든 고양이와 강아지 object들은 .한살먹기() 라는 함수를 사용할 수 있다.

    (1) 한살먹기 함수는 강아지 class로부터 생성된 오브젝트가 사용하면
              콘솔창에 에러를 출력해주어야합니다.

    (2) 한살먹기 함수는 고양이 class로 부터 생성된 오브젝트가 사용하면
              현재 가지고있는 age 속성에 1을 더해주는 기능을 실행해야합니다.

    class 강아지{   constructor(type, color){
        this.type=type;
        this.color=color;
      }
      한살먹기(){
        if(this instanceof 고양이){
          this.age++;
        }else{
          console.log('error!!');
        }
      }
    }
    class 고양이 extends 강아지{
      constructor(type, color, age){
        super(type, color);
        this.age = age;
      }
    }
    var 강아지1 = new 강아지('말티즈', 'white');
    var 고양이1 = new 고양이('코숏', 'white', 5);
    강아지1.한살먹기(1); // error!!
    고양이1.한살먹기();
    console.log(고양이1.age); // 6

  • 4. get/set을 이용하여 간단한 게임 기능을 가진 오브젝트를 뽑는 class 만들기

    조건

    class 이름은 Unit

    모든 Unit의 인스턴스는 공격력, 체력 속성이 있으며 기본 공격력은 5,
              기본 체력은 100으로 설정되어 있어야 합니다.

    모든 Unit의 인스턴스는 전투력을 측정해주는 battlePoint라는 getter가 있습니다.

    console.log( 인스턴스.battlePoint ) 이렇게 사용하면
              현재 공격력과 체력을 더한 값을 콘솔창에 출력해주어야합니다.

    (3) 모든 Unit의 인스턴스는 heal이라는 setter가 있습니다.

    인스턴스.heal = 50 이렇게 사용하면 체력 속성이 50 증가해야합니다.

    class Unit{
      constructor(){
        this.attack=5;
        this.hp=100;
      }
      get battlePoint(){
        return this.attack+this.hp;
      }
      set heal(addVal){
        this.hp = this.hp + addVal;
      }
    }
    var player1 = new Unit();
    console.log(player1.battlePoint); //105
    player1.heal=50;
    console.log(player1.hp); //150

  • 5. get/set을 이용하여 구현하기2

    var data = {
      odd : [],
      even : []
    }

    (1) data 오브젝트안에 setter 역할 함수를 하나 만들어보십시오.

    setter 함수에 1,2,3,4 이렇게 아무 자연수나 파라미터로 입력하면
              홀수는 odd, 짝수는 even 이라는 속성에 array 형태로 저장되어야합니다.

    var data ={
      odd:[],
      even:[],
      save(...num){
        for(let i=0; i< num.length; i+=1){
          if(num[i]%2===0){
            this.even.push(num[i]);
          }else{
            this.odd.push(num[i]);
          }
        }
      }
    }
    data.save(3,7,4);
    console.log(data.odd); //[3, 7]
    console.log(data.even); //[4]

    (2) data 오브젝트안에 getter 역할 함수를 하나 만들어보십시오.

    getter 함수를 사용하면 odd, even에 저장된
              모든 데이터들이 숫자순으로 정렬되어 출력되어야합니다.

    var data ={
      odd:[],
      even:[],
      save(...num){
        for(let i=0; i< num.length; i+=1){
          if(num[i]%2===0){
            this.even.push(num[i]);
          }else{
            this.odd.push(num[i]);
          }
        }
      },
      order(){
        return console.log([...this.odd, ...this.even ].sort());
      }
    }
    data.save(7,1,6,4,3,1,8,8,9,3);
    data.order(); //[1, 1, 3, 3, 4, 6, 7, 8, 8, 9]
