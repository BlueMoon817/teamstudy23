6. ES5방식으로 쉽게 구현하는 상속기능

  • Object.create(프로토타입object)

    var 부모 = {name: 'Kim', age:50}
    // 프로토타입을 정의
    // 직접 가지고 있는 건 없지만, 부모의 속성을 사용할 수 있다.
    var 자식 = Object.create(부모);
    console.log(자식); {}
    console.log(자식.name); // Kim
    console.log(자식.age); // 50

    // 자식의 속성값을 변경하기
    // 자식이 age속성을 동적으로 추가하면서 독립된 값을 할당한다.
    자식.age =20;
    console.log(자식.age); // 20
    console.log(부모.age); // 50

    // 손자에 자식을 상속하기
    var 손자 = Object.create(자식);
    // 부모에 name이 있으므로 자식을 거쳐서 부모의 값을 가져온다.
    console.log(손자.name); // Kim
    // 바로 위의 자식.age를 가져온다.
    console.log(손자.age); // 20
