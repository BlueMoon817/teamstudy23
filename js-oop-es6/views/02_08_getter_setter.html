8. getter, setter

  • getter, setter 사용이유

    함수를 만들어 object 데이터를 다루는 이유

    데이터를 동적으로 수정해주는 처리가 필요할 때

    데이터변경시 예외처리에 용이

    var 사람 = {
      name : 'Park',
      age:30,
      nextAge(){
        // 내년의 age
        return this.age+1
      },
      setAge(나이){
        // 데이터를 수정해주는 함수
        // 나이를 변경하는 함수
        // 데이터타입이 잘못들어왔을 때 parseInt로 숫자타입으로 변경해줄 수 있다.
        this.age = parseInt(나이);
      }
    }

    사람.nextAge(); // 31
    사람.setAge(20);
    사람.age;// 20

  • ES5 부터 추가된 문법

    set, get 키워드를 사용하여 함수를 만들면 호출시 소괄호를 생략할 수 있다.

    set은 데이터를 변경하는 함수에 사용

    setter 함수는 파라미터가 1개가 있어야 한다.

    get은 데이터를 불러오는 함수에 사용

    getter 함수는 return이 꼭 필요

    getter 함수는 파라미터가 없어야 함

    var 사람 = {
      name : 'Park',
      age:30,
      get nextAge(){
        return this.age+1
      },
      set setAge(나이){
        this.age = parseInt(나이);
      }
    }

    // 호출
    사람.setAge=20;
    사람.nextAge;

  • class에서 사용하는 get/set

  • class에서 사용하는 get/set

    class 사람 {
      constructor(){
        this.name='Park';
        this.age=20;
      }
      get nextAge(){
        return this.age + 1
      }
      set setAge(나이){
        this.age = 나이
      }
    }

    var 사람1 = new 사람();
    사람1.nextAge; // 21
    사람1.setAge=200
    console.log(사람1.age); // 201
