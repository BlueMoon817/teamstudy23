4. Destructuring 문법

  • 배열 데이터를 전부 변수에 담으려면?

    var arr = [2,3,4];
    var [a,b,c] = [2,3,4];
    //a=2,b=3,c=4 : 배열의 인자와 변수가 1:1 대응하여 생성 및 할당된다.
    var [a,b,c=10] = [2,3];
    // 선언할 변수에 기본값(파라메터)을 따로 줄 수 있다.
    // 배열의 요소 개수가 만들 변수 개수보다 부족하면 변수에 undefined가 할당된다.

  • Object 데이터를 꺼내 변수에 담으려면?

    var obj = {name:"Mun", age:30};
    var name= obj.name;
    var age = obj.age;
    // 객체의 키와 동일한 변수명을 가진 변수에 동일한 값을 변수 생성하면서 할당하고 싶을 경우
    var{name, age} = {name: "Mun", age:30};
    // 기본 파라메터 지정 가능
    var{name, age=24} = {name: "Mun"};
    var{name:이름, age=24} = {name: "Mun"};
    // 변수명이 name이 아닌 이름 이라는 변수로 만들고 싶다면 위와 같이 ': 변수명' 을 명시해준다. var{name:이름='park', age=24} = {name: "Mun"};
    // 다른 변수명에 기본 값을 지정하고 싶을 경우 위와 같이 작성한다.
    var name="Mun";
    var age = 30;
    var obj = {name :name, age:age};
    // 키와 변수 들어온 값이 같은이름이면 축약가능하다.
    var obj = {name, age};

  • 함수 파라미터 만들떄도 destructuring 문법 사용가능

    var obj = {name:"Mun", age:30}; function 함수(파라미터){ console.log(파라미터); } 함수(obj); function 함수({name , age}){ console.log(name); //Mun console.log(age); // 30 } 함수({name:"Mun", age:30}); function 함수([name , age]){ console.log(name);// 1 console.log(age); // 2 } 함수([1,2]);
