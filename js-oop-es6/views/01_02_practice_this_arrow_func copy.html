  • 연습문제1 : 간단한 메소드 만들기

    주어진 객체에 메소드를 추가하여 "안녕 나는 손흥민"을 출력하기

    var 사람 = {
      name: '손흥민'
    }
    사람.sayHi(); // 안녕 나는 손흥민
    var 사람 = {
      name: '손흥민',
      sayHi: function(){
        console.log(`안녕 나는 ${this.name}`);
      }
    }
    사람.sayHi(); // 안녕 나는 손흥민

  • 연습문제3 : setTimeout 사용하기

    버튼을 클릭시, 1초후에 '버튼이에요' 콘솔에 출력시키기


    <button id="버튼">버튼이에요</button>
    document.getElementID('버튼').addEventListener('click', function(){
      console.log(this.innerHTML);
    });
    <button id="버튼">버튼이에요</button>
    document.getElementID('버튼').addEventListener('click', function(){
      setTimeout(()=>{
         console.log(this.innerHTML);
      }, 1000)
    });
